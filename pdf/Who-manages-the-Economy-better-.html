df_GDP <- read.csv("API_NY.GDP.MKTP.KD.ZG_DS2_en_csv_v2_5454826.csv")
USA_GDP <- df_GDP[df_GDP$Country.Code == "USA", ]
USA_GDP <- USA_GDP[ , -1:-5]
USA_GDP <- melt(USA_GDP)
## Using X2022 as id variables
USA_GDP <- USA_GDP %>%
  rename("Party" = "X2022", "Year" = "variable", "GDP Growth" = "value")
USA_GDP$Year <- gsub("X", "", USA_GDP$Year)

USA_GDP$Year <- as.numeric(USA_GDP$Year)
USA_GDP$Party <- c("D", "D", "D", "D", "D", "D", "D", "D", "R", "R", "R", "R", "R", "R", "R", "R", "D", "D", "D","D", "R", "R", "R", "R", "R", "R", "R", "R", "R", "R", "R", "R", "D", "D", "D", "D", "D", "D", "D", "D", "R", "R", "R", "R", "R", "R", "R", "R", "D", "D", "D", "D", "D", "D", "D", "D", "R", "R", "R", "R", "D")
ggplot(USA_GDP, aes(x=Year, y=`GDP Growth`)) +
  geom_line() +
  geom_rect(aes(xmin = Year, xmax = dplyr::lead(Year), ymin = -3, ymax = Inf, fill = Party), alpha = 0.3) +
  theme_minimal() +
  scale_y_continuous(labels = function(x) paste0(x, "%")) +
  scale_fill_manual(values=c("blue", "red")) +
  ggtitle("GDP Growth by Party of President")
