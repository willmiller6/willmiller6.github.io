anes_econ <- data.frame(anes_full$VCF0004, anes_full$VCF0870, anes_full$VCF0872)
anes_econ <- na.omit(anes_econ)
anes_econ <- subset(anes_econ, anes_econ$anes_full.VCF0870 <= 5 & anes_econ$anes_full.VCF0872 <= 5)
anes_econ <- anes_econ %>%
  rename(year = anes_full.VCF0004, economy_last_year = anes_full.VCF0870, economy_next_year = anes_full.VCF0872)
anes_econ$economy_last_year <- recode(anes_econ$economy_last_year, "5" = 0, "3" = 5, "1" = 10)

anes_econ$economy_next_year <- recode(anes_econ$economy_next_year, "5" = 0, "3" = 5, "1" = 10)
anes_econ <- anes_econ %>%
  group_by(year) %>%
  summarise(mean_econ_last_year = mean(economy_last_year), mean_econ_next_year = mean(economy_next_year))
anes_econ$GDP_growth <- c(-0.3, -1.8, 7.2, 3.5, 4.2, 1.9, 3.5, 4.0, 3.8, 4.5, 4.1, 3.9, 0.1, 2.3, 1.7, -2.8)
ggplot(anes_econ, aes(x=year)) +
  geom_line(aes(y=mean_econ_last_year), color="darkred") +
  geom_line(aes(y=mean_econ_next_year), color="darkblue") +
  geom_line(aes(y=GDP_growth), color="darkgreen") +
  theme_minimal() +
  xlab("Year") +
  ylab("Economic Indicators") +
  ggtitle("Economic Perceptions vs. Economic Reality: 1980 - 2020")+
